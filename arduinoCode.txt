#include <Servo.h>

Servo myservo;  

int pos = 0;  

void setup() {
  myservo.attach(3);  
}

void loop() {
  for (pos = 0; pos <= 180; pos += 1) {

    myservo.write(pos);              
    delay(15);                       
    if( pos > 45) {
       digitalWrite(6, HIGH);
    }
    if( pos > 90) {
       digitalWrite(5, HIGH);
    }
    if( pos > 135) {
       digitalWrite(4, HIGH);
    }
  }
  for (pos = 180; pos >= 0; pos -= 1) { 
    myservo.write(pos);             
    delay(15);                       
    if( pos < 45) {
       digitalWrite(6, LOW);
    }
    if( pos < 90) {
       digitalWrite(5, LOW);
    }
    if( pos < 135) {
       digitalWrite(4, LOW);
    }
  }
}
